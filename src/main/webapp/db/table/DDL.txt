CREATE TABLE BOOKEY.TBL_CALENDAR
(
  DATEID VARCHAR2(8) PRIMARY KEY,
  DT DATE NOT NULL,
  DAYNUM NUMBER(1) NOT NULL,
  DAYCHA VARCHAR2(6) NOT NULL,
  YYYY VARCHAR2(4) NOT NULL,
  MM VARCHAR2(2) NOT NULL,
  DD VARCHAR2(2) NOT NULL,
  ISDAYOFF VARCHAR2(1),
  REMARK VARCHAR2(100),
  CREATED_DATE DATE DEFAULT SYSDATE,
  CREATED_USER  VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE DATE
  UPDATED_USER  VARCHAR2(100)
)


CREATE TABLE BOOKEY.TBL_CATEGORY
(
  CATGID VARCHAR2(8) PRIMARY KEY,
  CATG01 VARCHAR2(100),
  CATG02 VARCHAR2(100),
  CATG03 VARCHAR2(100),
  REMARK01 VARCHAR2(100),
  REMARK02 VARCHAR2(100),
  REMARK03 VARCHAR2(100),
  CREATED_DATE DATE DEFAULT SYSDATE,
  CREATED_USER  VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE DATE
  UPDATED_USER  VARCHAR2(100)
)


CREATE TABLE TBL_KEYWORD
(
  ID NUMBER(8) PRIMARY KEY,
  BOOKID  VARCHAR2(8),
  KEYWORD VARCHAR2(100),
  REMARK  VARCHAR2(500),
  CREATED_DATE  DATE DEFAULT SYSDATE,
  CREATED_USER  VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE DATE
  UPDATED_USER  VARCHAR2(100)
)

CREATE TABLE TBL_USER
(
  USERID  VARCHAR2(100) PRIMARY KEY,
  USERPW  VARCHAR2(100) NOT NULL,
  EMAIL   VARCHAR2(100) NOT NULL,
  NAME    VARCHAR2(100) NOT NULL,
  ADDRESS VARCHAR2(500) NOT NULL,
  IS_OPEN_TO_MARKETING  VARCHAR2(1) DEFAULT 'Y',
  BIRTHDAY  DATE NOT NULL,
  GENDER    VARCHAR2(1) NOT NULL,
  RANK      VARCHAR2(1) DEFAULT '1',
  AUTHNUM   VARCHAR2(7),
  CREATED_DATE  DATE  DEFAULT SYSDATE,
  CREATED_USER  VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE  DATE,
  UPDATED_USER  VARCHAR2(100)
);


CREATE TABLE TBL_MASTER
(
  TABLE_NAME  VARCHAR(100) NOT NULL,
  COLUMN_NAME VARCHAR2(100) NOT NULL,
  CODE        VARCHAR2(100) NOT NULL,
  VALUE       VARCHAR2(500),
  REMARK      VARCHAR2(500),
  CREATED_DATE  DATE  DEFAULT SYSDATE,
  CREATED_USER  VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE  DATE,
  UPDATED_USER  VARCHAR2(100),
  PRIMARY KEY(TABLE_NAME, COLUMN_NAME, CODE)
)
;


CREATE TABLE TBL_BOOK
(
  BOOKID    VARCHAR2(8) PRIMARY KEY,
  CATGID    VARCHAR2(8),
  BOOKNM    VARCHAR2(100) NOT NULL,
  PUBLISHER VARCHAR2(100) NOT NULL,
  AUTHOR    VARCHAR2(100) NOT NULL,
  PUBLISHED_DATE  DATE,
  STATUS        VARCHAR2(1) DEFAULT 'A',
  APPERANCE     VARCHAR2(1),
  IMAGE_FILE_NAME VARCHAR2(100),
  CREATED_DATE  DATE DEFAULT SYSDATE,
  CREATED_USER  VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE  DATE,
  UPDATED_USER  VARCHAR2(100)
);

CREATE TABLE TBL_RENTAL (

    RETURN_DATE DATE,

    QUEUE NUMBER,
    CREATED_DATE DATE DEFAULT SYSDATE,
    CREATED_USER VARCHAR2(100) DEFAULT 'MANUAL',
    UPDATED_DATE  DATE,

CREATE TABLE TBL_LOCATION
(
  LOCATION_ID VARCHAR2(8) PRIMARY KEY,
  BOOKID      VARCHAR2(8) NOT NULL,
  PLACE_NO    VARCHAR2(8) NOT NULL,
  SHELF_NO    VARCHAR2(8) NOT NULL,
  ROW_NO      VARCHAR2(8) NOT NULL,
  CREATED_DATE DATE DEFAULT SYSDATE,
  CREATED_USER VARCHAR2(100) DEFAULT 'MANUAL',
  UPDATED_DATE DATE,
  UPDATED_USER VARCHAR2(100),
  CONSTRAINT FK_BOOK FOREIGN KEY (BOOKID) REFERENCES TBL_BOOK(BOOKID)
)
